15.3 이분산성 

import statsmodels.stats.api as sms
resids = bos_y_test - rfr.predict(bos_X_test)
hb = sms.het_breuschpagan(resids, bos_X_test)
labels = [
  "Lagrange multiplier statistic",
  "p-value",
  "f-value",
  "f p-value",
]

for name, num in zip(labels, hb):
  print(f"{name}: {num:.2}")

p-value: 0.00036
f-value: 3.3
