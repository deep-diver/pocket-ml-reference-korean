import numpy as np

def correlated_columns(df, threshold=0.95):
  return (
    df.corr().pipe(
      lambda df1: pd.DataFrame(
        np.tril(df1, k=-1), columns=df.columns, index=df.columns
      )
    )
    .stack()
    .rename("pearson")
    .pipe(
      lambda s: s[
        s.abs() > threshold
      ].reset_index()
    )
    .query("level_0 not in level_1")
  )

correlated_columns(X)
