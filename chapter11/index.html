from yellowbrick.model_selection import (
  ValidationCurve
)

fig, ax = plt.subplots(figsize=(6, 4))
vc_viz = ValidationCurve(
  RandomForestClassifier(n_estimators=100),
  param_name="max_depth",
  param_range=np.arange(1, 11),
  cv=10,
